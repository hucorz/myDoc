    R¶

    基本操作¶

    ?funcname      # 打开函数帮助文件
ls()           # 查看所有对象
rm(x, y)       # 删除对象
rm(list=ls())  # 删除所有对象

set.seed(n)    # 指定随机数种子

q()            # 退出

    Matrix¶

    向量¶

    x = c(1,2,3,4)        # connect 建立数值向量
seq(a, b, length=n) # [a,b]之间等距的n个元素的向量，也可以直接 a:b
rnorm(n)              # 产生长度为n的随机正态分布向量，如果指定了随机数种子那每次的数会一样

length(x)             # 返回向量长度
mean(x)               # 向量均值 
sd(x)                 # 标准差
var(x)                # 向量方差
cor(x,y)              # 计算2向量的相关系数

    Matrix¶

    matrix(data=NA, nrow=1, ncol=1, byrow=FALSE)
# eg: x = (data=c(1,2,3,4),2,2) 只写前三个参数名可以省略
# 默认按列填写matrix, byrow=TRUE会按行填写

dim(A)   # 输出维数
sqrt(x)  # 每个元素开根
x^2      # 每个元素开方

    索引数据¶

    A[1, 2]               # 指定单元格
A[c(x1,x2), c(y1,y2)] # 指定某些列和某些行
A[x1:x2, y1:y2]       # 指定行与列的范围
A[x1:x2,]             # 指定行，不指定列

A[-c(x1,x2)]          # 指定不包含的数据

    数据处理¶

    a = as.factor(a)   # 如果a的数据是离散的转换成一个定性的变量
# eg:  
# > a=[1,2,2,3,1,2,3,2,1,3] 
# > a=as.factor(a)
# > print(a)
# [1] 1 2 3 1 2 1 3 2 1
# Levels: 1 2 3

summary(a)    # 对指定数据集中的每个变量计算统计信息(最大值，最小值...)
summary(name) # 也可以指定某个具体的属性名

    图像¶

    建立图像¶

    # 如果x是定性(离散)变量会自动画箱线图
plot(x, y, xlim=(x1, x2), xlab="", main="")   # main: title

hist(x)                      # 直方图,纵坐标是落在区间内元素的个数

# 通过 IO 数据作图
plot(a$name1, a$name2)               # 数据集和变量名间加$
attach(a)                            # 也可以指定数据集
plot(name1, name2)

    等高线图¶

    outer(x, y, FUN="*")  # 建立一个len(x)*len(y)维的矩阵，fun默认是*,即 xi*yj
contour(x, y, z, nlevels=10)        # 等高线 nlevels:等高线数量
image(x, y, z, xlim, xlab)            # 热地图
persp(x, y, z, xlim, xlab, theta, phi)# 三维图 theta/phi 控制图形查看角度eg:persp(x, y, z, theta = 30, phi = 20)

    保存图像¶

    pdf("test.pdf") # 用这个文件保存图像
plot(x, y)      
dev.off()       # 指示创建图像的工作结束

    读入数据¶

    a = read.table("filepath", header=T, na.strings="") # 读入文本文件,指定表头和缺失值
a = read.csv()                                      # 读入csv(','分隔符文件)
a = na.omit(a)                       # 直接删去有缺失值的行

fix(a)                               # 展示数据
name(a)                              # 查看表头(变量名)
